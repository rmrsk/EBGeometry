219  template <class T>
220  inline T
221  EdgeT<T>::signedDistance(const Vec3& a_x0) const noexcept
222  {
223  // Project point to edge.
224  const T t = this->projectPointToEdge(a_x0);
225 
226  T retval;
227  if (t <= 0.0) {
228  // Closest point is the starting vertex
229  retval = this->getVertex()->signedDistance(a_x0);
230  }
231  else if (t >= 1.0) {
232  // Closest point is the end vertex
233  retval = this->getOtherVertex()->signedDistance(a_x0);
234  }
235  else {
236  // Closest point is the edge itself.
237  const Vec3 x2x1 = this->getX2X1();
238  const Vec3 linePoint = m_vertex->getPosition() + t * x2x1;
239  const Vec3 delta = a_x0 - linePoint;
240  const T dot = m_normal.dot(delta);
241 
242  const int sgn = (dot > 0.0) ? 1 : -1;
243 
244  retval = sgn * delta.length();
245  }
246 
247  return retval;
248  }

252  EdgeT<T>::unsignedDistance2(const Vec3& a_x0) const noexcept
