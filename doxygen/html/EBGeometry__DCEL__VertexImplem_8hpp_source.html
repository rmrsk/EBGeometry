111  VertexT<T>::computeVertexNormalAverage(const std::vector<FacePtr>& a_faces) noexcept
112  {
113  m_normal = Vec3::zero();
114 
115  // TLDR: We simply compute the sum of the normal vectors for each face in
116  // a_faces and then normalize. This
117  // will yield an "average" of the normal vectors of the faces
118  // circulating this vertex.

121  }
122 
123  this->normalizeNormalVector();

133  template <class T>
134  inline void
135  VertexT<T>::computeVertexNormalAngleWeighted(const std::vector<FacePtr>& a_faces) noexcept
136  {
137  m_normal = Vec3::zero();
138 
139  // This routine computes the pseudonormal from pseudnormal algorithm from
140  // Baerentzen and Aanes in "Signed distance computation using the angle
141  // weighted pseudonormal" (DOI: 10.1109/TVCG.2005.49). This algorithm computes
142  // an average normal vector using the normal vectors of each face connected to
143  // this vertex, i.e. in the form
144  //
145  // n = sum(w * n(face))/sum(w)
146  //
147  // where w are weights for each face. This weight is given by the subtended
148  // angle of the face, which means the angle spanned by the incoming/outgoing
149  // edges of the face that pass through this vertex.
150  //
151  //
152  // The below code is more complicated than it looks. It happens because we
153  // want the two half edges that has the current vertex as a mutual vertex
154  // (i.e. the "incoming" and "outgoing" edges into this vertex). Normally we'd
155  // just iterate through edges, but if it happens that an input face is
156  // flipped, this will result in infinite iteration. Instead, we have stored
157  // the pointers to each face connected to this vertex. We look through each
158  // face to find the endpoints of the edges the have the current vertex as the
159  // common vertex, and then compute the subtended angle between those. Sigh...
160 
161  const VertexPtr& originVertex = m_outgoingEdge->getVertex(); // AKA 'this'
162 
163  for (const auto& f : a_faces) {
164 
165  std::vector<VertexPtr> inoutVertices(0);
166  for (EdgeIterator edgeIt(f->getHalfEdge()); edgeIt.ok(); ++edgeIt) {
167  const auto& e = edgeIt();
168 
169  const auto& v1 = e->getVertex();
170  const auto& v2 = e->getOtherVertex();
171 
172  if (v1 == originVertex || v2 == originVertex) {
173  if (v1 == originVertex) {
174  inoutVertices.emplace_back(v2);
175  }
176  else if (v2 == originVertex) {
177  inoutVertices.emplace_back(v1);
178  }
179  else {
180  std::cerr << "In file 'CD_DCELVertexImplem.H' function "
181  "vertexT<T>::computeVertexNormalAngleWeighted() - logic bust.\n";
182  }
183  }
184  }
185 
186  if (inoutVertices.size() != 2) {
187  std::cerr << "In file 'CD_DCELVertexImplem.H' function "
188  "vertexT<T>::computeVertexNormalAngleWeighted() - logic bust 2.\n";
189  }
190 
191  const Vec3& x0 = originVertex->getPosition();
192  const Vec3& x1 = inoutVertices[0]->getPosition();
193  const Vec3& x2 = inoutVertices[1]->getPosition();
194 
195  if (x0 == x1 || x0 == x2 || x1 == x2) {
196  std::cerr << "In file 'CD_DCELVertexImplem.H' function "
197  "vertexT<T>::computeVertexNormalAngleWeighted() - logic bust 3.\n";
198  }
199 
200  Vec3 v1 = x1 - x0;
201  Vec3 v2 = x2 - x0;
202 
203  v1 = v1 / v1.length();
204  v2 = v2 / v2.length();
205 
206  const Vec3& norm = f->getNormal();
207 
208  const T alpha = acos(v1.dot(v2));
209 
210  m_normal += alpha * norm;
211  }
212 
213  this->normalizeNormalVector();
214  }

282  {
