92 template <class T>
93 inline int
94 Polygon2D<T>::computeWindingNumber(const Vec2& P) const noexcept
95 {
96  int wn = 0; // the winding number counter
97 
98  const size_t N = m_points.size();
99 
100  auto isLeft = [](const Vec2& P0, const Vec2& P1, const Vec2& P2) {
101  return (P1.x - P0.x) * (P2.y - P0.y) - (P2.x - P0.x) * (P1.y - P0.y);
102  };
103 
104  // loop through all edges of the polygon
105  for (size_t i = 0; i < N; i++) { // edge from V[i] to V[i+1]
106 
107  const Vec2& P1 = m_points[i];
108  const Vec2& P2 = m_points[(i + 1) % N];
109 
110  const T res = isLeft(P1, P2, P);
111 
112  if (P1.y <= P.y) { // start y <= P.y
113  if (P2.y > P.y) // an upward crossing
114  if (res > 0.) // P left of edge
115  ++wn; // have a valid up intersect
116  }
117  else { // start y > P.y (no test needed)
118  if (P2.y <= P.y) // a downward crossing
119  if (res < 0.) // P right of edge
120  --wn; // have a valid down intersect
121  }
122  }
123 
124  return wn;
125 }

127 template <class T>
128 inline size_t
129 Polygon2D<T>::computeCrossingNumber(const Vec2& P) const noexcept
130 {
131  size_t cn = 0;
132 
133  const size_t N = m_points.size();
134 
135  for (size_t i = 0; i < N; i++) { // edge from V[i] to V[i+1]
136  const Vec2& P1 = m_points[i];
137  const Vec2& P2 = m_points[(i + 1) % N];
138 
139  const bool upwardCrossing = (P1.y <= P.y) && (P2.y > P.y);
140  const bool downwardCrossing = (P1.y > P.y) && (P2.y <= P.y);
141 
142  if (upwardCrossing || downwardCrossing) {
143  const T t = (P.y - P1.y) / (P2.y - P1.y);
144 
145  if (P.x < P1.x + t * (P2.x - P1.x)) { // P.x < intersect
146  cn += 1; // a valid crossing of y=P.y right of P.x
147  }
148  }
149  }
150 
151  return cn;
152 }

154 template <class T>
155 inline T
156 Polygon2D<T>::computeSubtendedAngle(const Vec2& p) const noexcept
157 {
158  T sumTheta = 0.0;
159 
160  const size_t N = m_points.size();
161 
162  for (size_t i = 0; i < N; i++) {
163  const Vec2 p1 = m_points[i] - p;
164  const Vec2 p2 = m_points[(i + 1) % N] - p;
165 
166  const T theta1 = atan2(p1.y, p1.x);
167  const T theta2 = atan2(p2.y, p2.x);
168 
169  T dTheta = theta2 - theta1;
170 
171  while (dTheta > M_PI)
172  dTheta -= 2.0 * M_PI;
173  while (dTheta < -M_PI)
174  dTheta += 2.0 * M_PI;
175 
176  sumTheta += dTheta;
177  }
178 
179  return sumTheta;
180 }
