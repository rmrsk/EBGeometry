106  template <class T>
107  inline void
108  FaceT<T>::computeArea() noexcept
109  {
110  m_area = 0.0;
111 
112  // This computes the area of any N-side polygon.
113  const auto vertices = this->gatherVertices();
114 
115  for (size_t i = 0; i < vertices.size() - 1; i++) {
116  const auto& v1 = vertices[i]->getPosition();
117  const auto& v2 = vertices[i + 1]->getPosition();
118  m_area += m_normal.dot(v2.cross(v1));
119  }
120 
121  m_area = 0.5 * std::abs(m_area);
122  }

139  template <class T>
140  inline void
141  FaceT<T>::computeNormal() noexcept
142  {
143  const auto vertices = this->gatherVertices();
144 
145  const size_t N = vertices.size();
146 
147  // To compute the normal vector we find three vertices in this polygon face.
148  // They span a plane, and we just compute the normal vector of that plane.
149  for (size_t i = 0; i < N; i++) {
150  const auto& x0 = vertices[i]->getPosition();
151  const auto& x1 = vertices[(i + 1) % N]->getPosition();
152  const auto& x2 = vertices[(i + 2) % N]->getPosition();
153 
154  m_normal = (x2 - x1).cross(x2 - x0);
155 
156  if (m_normal.length() > 0.0) {
157  break; // Found one.
158  }
159  }
160 
161  this->normalizeNormalVector();
162  }

315  template <class T>
316  inline T
317  FaceT<T>::signedDistance(const Vec3& a_x0) const noexcept
318  {
319  T retval = std::numeric_limits<T>::infinity();
320 
321  const bool inside = this->isPointInsideFace(a_x0);
322 
323  if (inside) {
324  retval = m_normal.dot(a_x0 - m_centroid);
325  }
326  else {
327  for (const auto& e : m_edges) {
328  const T curDist = e->signedDistance(a_x0);
329 
330  retval = (std::abs(curDist) < std::abs(retval)) ? curDist : retval;
331  }
332  }
333 
334  return retval;
335  }
