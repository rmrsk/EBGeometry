130  template <class X, size_t K>
131  auto equalCounts = [](const std::vector<X>& a_primitives) -> std::array<std::vector<X>, K> {
132  int length = a_primitives.size() / K;
133  int remain = a_primitives.size() % K;
134 
135  int begin = 0;
136  int end = 0;
137 
138  std::array<std::vector<X>, K> chunks;
139 
140  for (size_t k = 0; k < K; k++) {
141  end += (remain > 0) ? length + 1 : length;
142  remain--;
143 
144  chunks[k] = std::vector<X>(a_primitives.begin() + begin, a_primitives.begin() + end);
145 
146  begin = end;
147  }
148 
149  return chunks;
150  };

156  template <class T, class P, class BV, size_t K>
157  auto PrimitiveCentroidPartitioner =
158  [](const PrimAndBVListT<P, BV>& a_primsAndBVs) -> std::array<PrimAndBVListT<P, BV>, K> {
159  Vec3T<T> lo = Vec3T<T>::max();
160  Vec3T<T> hi = -Vec3T<T>::max();
161 
162  for (const auto& pbv : a_primsAndBVs) {
163  lo = min(lo, pbv.first->getCentroid());
164  hi = max(hi, pbv.first->getCentroid());
165  }
166 
167  const size_t splitDir = (hi - lo).maxDir(true);
168 
169  // Sort the primitives along the above coordinate direction.
170  PrimAndBVListT<P, BV> sortedPrimsAndBVs(a_primsAndBVs);
171 
172  std::sort(sortedPrimsAndBVs.begin(),
173  sortedPrimsAndBVs.end(),
174  [splitDir](const PrimAndBV<P, BV>& pbv1, const PrimAndBV<P, BV>& pbv2) -> bool {
175  return pbv1.first->getCentroid(splitDir) < pbv2.first->getCentroid(splitDir);
176  });
177 
178  return BVH::equalCounts<PrimAndBV<P, BV>, K>(sortedPrimsAndBVs);
179  };

187  Vec3T<T> lo = Vec3T<T>::max();
