131  template <class X, size_t K>
132  auto equalCounts = [](const std::vector<X>& a_primitives) -> std::array<std::vector<X>, K> {
133  int length = a_primitives.size() / K;
134  int remain = a_primitives.size() % K;
135 
136  int begin = 0;
137  int end = 0;
138 
139  std::array<std::vector<X>, K> chunks;
140 
141  for (size_t k = 0; k < K; k++) {
142  end += (remain > 0) ? length + 1 : length;
143  remain--;
144 
145  chunks[k] = std::vector<X>(a_primitives.begin() + begin, a_primitives.begin() + end);
146 
147  begin = end;
148  }
149 
150  return chunks;
151  };

157  template <class T, class P, class BV, size_t K>
158  auto PrimitiveCentroidPartitioner =
159  [](const PrimAndBVListT<P, BV>& a_primsAndBVs) -> std::array<PrimAndBVListT<P, BV>, K> {
160  Vec3T<T> lo = Vec3T<T>::max();
161  Vec3T<T> hi = -Vec3T<T>::max();
162 
163  for (const auto& pbv : a_primsAndBVs) {
164  lo = min(lo, pbv.first->getCentroid());
165  hi = max(hi, pbv.first->getCentroid());
166  }
167 
168  const size_t splitDir = (hi - lo).maxDir(true);
169 
170  // Sort the primitives along the above coordinate direction.
171  PrimAndBVListT<P, BV> sortedPrimsAndBVs(a_primsAndBVs);
172 
173  std::sort(sortedPrimsAndBVs.begin(),
174  sortedPrimsAndBVs.end(),
175  [splitDir](const PrimAndBV<P, BV>& pbv1, const PrimAndBV<P, BV>& pbv2) -> bool {
176  return pbv1.first->getCentroid(splitDir) < pbv2.first->getCentroid(splitDir);
177  });
178 
179  return BVH::equalCounts<PrimAndBV<P, BV>, K>(sortedPrimsAndBVs);
180  };
