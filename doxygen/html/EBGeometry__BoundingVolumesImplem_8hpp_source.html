45  template <class T>
46  BoundingSphereT<T>::BoundingSphereT(const std::vector<BoundingSphereT<T>>& a_otherSpheres)
47  {
48 
49  // TLDR: Spheres enclosing other spheres is a difficult problem, but a sphere
50  // enclosing a set of points is simpler. For each
51  // input sphere we create a set of points representing the lo/hicorners
52  // of an axis-aligned bounding box that encloses the sphere. We then
53  // compute the bounding sphere from this set of points.
54  std::vector<Vec3T<T>> points;
55  for (const auto& sphere : a_otherSpheres) {
56  const T& radius = sphere.getRadius();
57  const Vec3T<T>& center = sphere.getCentroid();
58 
59  points.emplace_back(center + radius * Vec3T<T>::one());
60  points.emplace_back(center - radius * Vec3T<T>::one());
61  }
62 
63  this->define(points, BoundingSphereT<T>::BoundingVolumeAlgorithm::Ritter);
64  }

134  inline void
135  BoundingSphereT<T>::buildRitter(const std::vector<Vec3T<P>>& a_points) noexcept
136  {
137  m_radius = 0.0;
138  m_center = Vec3::zero();
139 
140  constexpr T half = 0.5;
141 
142  constexpr size_t DIM = 3;
143 
144  // INITIAL PASS
145  std::vector<Vec3> min_coord(DIM, a_points[0]); // [0] = Minimum x, [1] = Minimum y, [2] = Minimum z
146  std::vector<Vec3> max_coord(DIM, a_points[0]);
147 
148  for (size_t i = 1; i < a_points.size(); i++) {
149  for (size_t dir = 0; dir < DIM; dir++) {
150  Vec3& min = min_coord[dir];
151  Vec3& max = max_coord[dir];
152 
153  if (a_points[i][dir] < min[dir]) {
154  min = a_points[i];
155  }
156  if (a_points[i][dir] > max[dir]) {
157  max = a_points[i];
158  }
159  }
160  }
161 
162  T dist = -1;
163  Vec3 v, p1, p2;
164  for (size_t dir = 0; dir < DIM; dir++) {
165  const T len = (max_coord[dir] - min_coord[dir]).length();
166  if (len > dist) {
167  dist = len;
168  p1 = min_coord[dir];
169  p2 = max_coord[dir];
170  }
171  }
172 
173  // m_center = half*(p1+p2);
174  m_center = (p1 + p2) * half;
175  m_radius = half * (p2 - p1).length();
176 
177  // SECOND PASS
178  for (size_t i = 0; i < a_points.size(); i++) {
179  dist = (a_points[i] - m_center).length() - m_radius;
180  if (dist > 0.0) { // Point lies outside
181  v = a_points[i] - m_center;
182  p1 = a_points[i];
183  p2 = m_center - m_radius * v / v.length();
184 
185  m_center = half * (p2 + p1);
186  m_radius = half * (p2 - p1).length();
187  }
188  }
189 
190  // Ritter algorithm is very coarse and does not give an exact result anyways.
191  // Grow the dimension for safety.
192  m_radius *= (1.0 + 1E-2);
193  }

196  inline T
197  BoundingSphereT<T>::getOverlappingVolume(const BoundingSphereT<T>& a_other) const noexcept
198  {
199  constexpr T zero = 0.0;
200 
201  T retval = zero;
202 
203  if (this->intersects(a_other)) {
204  const auto& r1 = m_radius;
205  const auto& r2 = a_other.getRadius();
206 
207  const auto d = (m_center - a_other.getCentroid()).length();
208 
209  retval =
210  M_PI / (12. * d) * (r1 + r2 - d) * (r1 + r2 - d) * (d * d + 2 * d * (r1 + r2) - 3 * (r1 - r2) * (r1 - r2));
211  }
212 
213  return retval;
214  }
