42 inline void
43 Parser::PLY<T>::readIntoDCEL(Mesh& a_mesh, const std::string a_filename)
44 {
45  std::ifstream filestream(a_filename);
46 
47  if (filestream.is_open()) {
48  std::vector<std::shared_ptr<Vertex>>& vertices = a_mesh.getVertices();
49  std::vector<std::shared_ptr<Edge>>& edges = a_mesh.getEdges();
50  std::vector<std::shared_ptr<Face>>& faces = a_mesh.getFaces();
51 
52  vertices.resize(0);
53  edges.resize(0);
54  faces.resize(0);
55 
56  size_t numVertices; // Number of vertices
57  size_t numFaces; // Number of faces
58 
59  Parser::PLY<T>::readHeaderASCII(numVertices, numFaces, filestream);
60  Parser::PLY<T>::readVerticesIntoDCEL(vertices, numVertices, filestream);
61  Parser::PLY<T>::readFacesIntoDCEL(faces, edges, vertices, numFaces, filestream);
62  Parser::PLY<T>::reconcilePairEdgesDCEL(edges);
63 
64  a_mesh.sanityCheck();
65 
66  filestream.close();
67 
68  a_mesh.reconcile(EBGeometry::DCEL::MeshT<T>::VertexNormalWeight::Angle);
69  }
70  else {
71  const std::string error = "Parser::PLY::readIntoDCEL - ERROR! Could not open file " + a_filename;
72  std::cerr << error + "\n";
73  }
74 }

77 inline void
78 Parser::PLY<T>::readHeaderASCII(size_t& a_numVertices, size_t& a_numFaces, std::ifstream& a_inputStream)
79 {
80 
81  std::string str1;
82  std::string str2;
83  std::string line;
84 
85  // Get number of vertices
86  a_inputStream.clear();
87  a_inputStream.seekg(0);
88  while (getline(a_inputStream, line)) {
89  std::stringstream sstream(line);
90  sstream >> str1 >> str2 >> a_numVertices;
91  if (str1 == "element" && str2 == "vertex") {
92  break;
93  }
94  }
95 
96  // Get number of faces
97  a_inputStream.clear();
98  a_inputStream.seekg(0);
99  while (getline(a_inputStream, line)) {
100  std::stringstream sstream(line);
101  sstream >> str1 >> str2 >> a_numFaces;
102  if (str1 == "element" && str2 == "face") {
103  break;
104  }
105  }
106 
107  // Find the line # containing "end_header" halt the input stream there
108  a_inputStream.clear();
109  a_inputStream.seekg(0);
110  while (getline(a_inputStream, line)) {
111  std::stringstream sstream(line);
112  sstream >> str1;
113  if (str1 == "end_header") {
114  break;
115  }
116  }

119 template <class T>
120 inline void
121 Parser::PLY<T>::readVerticesIntoDCEL(std::vector<std::shared_ptr<Vertex>>& a_vertices,
122  const size_t a_numVertices,
123  std::ifstream& a_inputStream)
124 {
125 
126  Vec3T<T> pos;
127  T& x = pos[0];
128  T& y = pos[1];
129  T& z = pos[2];
130 
131  Vec3T<T> norm;
132  T& nx = norm[0];
133  T& ny = norm[1];
134  T& nz = norm[2];
135 
136  size_t num = 0;
137 
138  std::string line;
139  while (std::getline(a_inputStream, line)) {
140  std::stringstream sstream(line);
141  sstream >> x >> y >> z >> nx >> ny >> nz;
142 
143  a_vertices.emplace_back(std::make_shared<Vertex>(pos, norm));
144 
145  // We have read all the vertices we should read. Exit now -- after this the
146  // inputStream will begin reading faces.
147  num++;
148  if (num == a_numVertices)
149  break;
150  }
151 }

153 template <class T>
154 inline void
155 Parser::PLY<T>::readFacesIntoDCEL(std::vector<std::shared_ptr<Face>>& a_faces,
156  std::vector<std::shared_ptr<Edge>>& a_edges,
157  const std::vector<std::shared_ptr<Vertex>>& a_vertices,
158  const size_t a_numFaces,
159  std::ifstream& a_inputStream)
160 {
161  size_t numVertices;
162  std::vector<size_t> vertexIndices;
163 
164  std::string line;
165  size_t counter = 0;
166  while (std::getline(a_inputStream, line)) {
167  counter++;
168 
169  std::stringstream sstream(line);
170 
171  sstream >> numVertices;
172  vertexIndices.resize(numVertices);
173  for (size_t i = 0; i < numVertices; i++) {
174  sstream >> vertexIndices[i];
175  }
176 
177  if (numVertices < 3)
178  std::cerr << "Parser::PLY::readFacesIntoDCEL - a face must have at least "
179  "three vertices!\n";
180 
181  // Get the vertices that make up this face.
182  std::vector<std::shared_ptr<Vertex>> curVertices;
183  for (size_t i = 0; i < numVertices; i++) {
184  const size_t vertexIndex = vertexIndices[i];
185  curVertices.emplace_back(a_vertices[vertexIndex]);
186  }
187 
188  // Build inside half edges and give each vertex an outgoing half edge. This
189  // may get overwritten later, but the outgoing edge is not unique so it
190  // doesn't matter.
191  std::vector<std::shared_ptr<Edge>> halfEdges;
192  for (const auto& v : curVertices) {
193  halfEdges.emplace_back(std::make_shared<Edge>(v));
194  v->setEdge(halfEdges.back());
195  }
196 
197  a_edges.insert(a_edges.end(), halfEdges.begin(), halfEdges.end());
198 
199  // Associate next/previous for the half edges inside the current face. Wish
200  // we had a circular iterator but this will have to do.
201  for (size_t i = 0; i < halfEdges.size(); i++) {
202  auto& curEdge = halfEdges[i];
203  auto& nextEdge = halfEdges[(i + 1) % halfEdges.size()];
204 
205  curEdge->setNextEdge(nextEdge);
206  nextEdge->setPreviousEdge(curEdge);
207  }
208 
209  // Construct a new face
210  a_faces.emplace_back(std::make_shared<Face>(halfEdges.front()));
211  auto& curFace = a_faces.back();
212 
213  // Half edges get a reference to the currently created face
214  for (auto& e : halfEdges) {
215  e->setFace(curFace);
216  }
217 
218  // Must give vertices access to all faces associated with them since PLY
219  // files do not give any edge association.
220  for (auto& v : curVertices) {
221  v->addFace(curFace);
222  }
223 
224  if (counter == a_numFaces)
225  break;
226  }
227 }

229 template <class T>
230 inline void
231 Parser::PLY<T>::reconcilePairEdgesDCEL(std::vector<std::shared_ptr<Edge>>& a_edges)
232 {
233  for (auto& curEdge : a_edges) {
234  const auto& nextEdge = curEdge->getNextEdge();
235 
236  const auto& vertexStart = curEdge->getVertex();
237  const auto& vertexEnd = nextEdge->getVertex();
238 
239  for (const auto& p : vertexStart->getFaces()) {
240  for (EdgeIterator edgeIt(*p); edgeIt.ok(); ++edgeIt) {
241  const auto& polyCurEdge = edgeIt();
242  const auto& polyNextEdge = polyCurEdge->getNextEdge();
243 
244  const auto& polyVertexStart = polyCurEdge->getVertex();
245  const auto& polyVertexEnd = polyNextEdge->getVertex();
246 
247  if (vertexStart == polyVertexEnd && polyVertexStart == vertexEnd) { // Found the pair edge
248  curEdge->setPairEdge(polyCurEdge);
249  polyCurEdge->setPairEdge(curEdge);
250  }
251  }
252  }
253  }
254 }
