129  EBGeometry::BVH::PartitionerT<EBGeometry::Dcel::FaceT<T>, BV, K> bvPartitioner = [](const PrimitiveList<T>& a_primitives) -> std::array<PrimitiveList<T>, K> {
130  Vec3T<T> lo = Vec3T<T>::max();
131  Vec3T<T> hi = -Vec3T<T>::max();
132

134  using P = std::pair<std::shared_ptr<const EBGeometry::Dcel::FaceT<T> >, BV>;
135 
136  std::vector<P> primsAndBVs;

140 
141  primsAndBVs.emplace_back(p, bv);
142 
143  lo = min(lo, bv.getCentroid());

148

150  std::sort(primsAndBVs.begin(), primsAndBVs.end(),
151  [splitDir](const P& p1, const P& p2){
152  return (p1.second).getCentroid()[splitDir] < (p2.second).getCentroid()[splitDir];
153  });
154

156  PrimitiveList<T> sortedPrimitives;
157  for (const auto& p : primsAndBVs){
158  sortedPrimitives.emplace_back(p.first);
159  }
160

170  template <class T, class BV, size_t K>
171  EBGeometry::BVH::PartitionerT<EBGeometry::Dcel::FaceT<T>, BV, K> centroidPartitioner = [](const PrimitiveList<T>& a_primitives) -> std::array<PrimitiveList<T>, K> {
172  Vec3T<T> lo = Vec3T<T>::max();
173  Vec3T<T> hi = -Vec3T<T>::max();
174  for (const auto& p : a_primitives){
175  lo = min(lo, p->getCentroid());
176  hi = max(hi, p->getCentroid());
177  }
178 
179  const size_t splitDir = (hi-lo).maxDir(true);
180  const T delta = (hi-lo)[splitDir]/K;
181 
182  std::array<T, K> boundsLo;
183  std::array<T, K> boundsHi;
184 
185  for (size_t k = 0; k < K; k++){
186  boundsLo[k] = lo[splitDir] + delta*k;
187  boundsHi[k] = lo[splitDir] + delta*(k+1);
188  }
189 
190  // Given coord, find the bin.
191  auto getBin = [&](const T& coord) -> size_t {
192  for (size_t k = 0; k < K; k++) {
193  if(coord >= boundsLo[k] && coord <= boundsHi[k]) {
194  return k;
195  }
196  }
197 
198  return K-1;
199  };
200 
201  // Put primitives in their respective bins.
202  std::array<PrimitiveList<T>, K> chunks;
203 
204  for (const auto& p : a_primitives){
205  const size_t k = getBin(p->getCentroid()[splitDir]);
206  chunks[k].push_back(p);
207  }
208 
209  // The centroid-based partitioner can end up with no primitives in one of the leaves (a rare case). Use a different partitioner in
210  // that case.
211  if(!(EBGeometry::Dcel::validChunks<T, K>(chunks))) {
212  chunks = EBGeometry::Dcel::chunkPartitioner<T, K>(a_primitives);
213  }
214 
215  return chunks;
216  };
