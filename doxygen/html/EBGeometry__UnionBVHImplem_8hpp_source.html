29 template <class T, class P, class BV, size_t K>
30 void
31 UnionBVH<T, P, BV, K>::buildTree(const std::vector<std::shared_ptr<P>>& a_distanceFunctions,
32  const BVConstructor& a_bvConstructor) noexcept
33 {
34  using PrimList = std::vector<std::shared_ptr<const P>>;
35  using BuilderNode = EBGeometry::BVH::NodeT<T, P, BV, K>;
36 
37  // This function is a partitioning function taking the input primitives and
38  // partitioning them into K subvolumes. Since the SDFs don't themselves have
39  // vertices, centroids, etc., we use their bounding volumes as criteria for
40  // subdividing them. We do this by computing the spatial centroid of each
41  // bounding volume that encloses the SDFs. We then do a spatial subdivision
42  // along the longest coordinate into K almost-equal chunks.
43  EBGeometry::BVH::PartitionerT<P, BV, K> partitioner =
44  [a_bvConstructor](const PrimList& a_primitives) -> std::array<PrimList, K> {
45  const size_t numPrimitives = a_primitives.size();
46 
47  if (numPrimitives < K) {
48  std::cerr << "UnionBVH<T, P, BV, K>::buildTree -- not enough primitives to "
49  "partition into K new nodes\n";
50  }
51 
52  // 1. Compute the bounding volume centroids for each input SDF.
53  std::vector<Vec3T<T>> bvCentroids;
54  for (const auto& prim : a_primitives) {
55  bvCentroids.emplace_back((a_bvConstructor(prim)).getCentroid());
56  }
57 
58  // 2. Figure out which coordinate direction has the longest/smallest extent.
59  // We split along the longest direction.
60  auto lo = Vec3T<T>::infinity();
61  auto hi = -Vec3T<T>::infinity();
62 
63  for (const auto& c : bvCentroids) {
64  lo = min(lo, c);
65  hi = max(hi, c);
66  }
67 
68  const size_t splitDir = (hi - lo).maxDir(true);
69 
70  // 3. Sort input primitives based on the centroid location of their bounding
71  // volumes. We do this by packing the SDFs and their BV centroids
72  // in a vector which we sort (I love C++).
73  using Primitive = std::shared_ptr<const P>;
74  using Centroid = Vec3T<T>;
75  using PC = std::pair<Primitive, Centroid>;
76 
77  // Vector pack.
78  std::vector<PC> primsAndCentroids;
79  for (size_t i = 0; i < a_primitives.size(); i++) {
80  primsAndCentroids.emplace_back(a_primitives[i], bvCentroids[i]);
81  }
82 
83  // Vector sort.
84  std::sort(primsAndCentroids.begin(), primsAndCentroids.end(), [splitDir](const PC& sdf1, const PC& sdf2) -> bool {
85  return sdf1.second[splitDir] < sdf2.second[splitDir];
86  });
87 
88  // Vector unpack. The input SDFs are not sorted based on their bounding
89  // volume centroids.
90  std::vector<Primitive> sortedPrimitives;
91  for (size_t i = 0; i < numPrimitives; i++) {
92  sortedPrimitives.emplace_back(primsAndCentroids[i].first);
93  }
94 
95  // 4. Figure out where along the PC vector we should do our spatial splits.
96  // We try to balance the chunks.
97  const size_t almostEqualChunkSize = numPrimitives / K;
98  size_t remainder = numPrimitives % K;
99 
100  std::array<size_t, K> startIndices;
101  std::array<size_t, K> endIndices;
102 
103  startIndices[0] = 0;
104  endIndices[K - 1] = numPrimitives;
105 
106  for (size_t i = 1; i < K; i++) {
107  startIndices[i] = startIndices[i - 1] + almostEqualChunkSize;
108 
109  if (remainder > 0) {
110  startIndices[i]++;
111  remainder--;
112  }
113  }
114 
115  for (size_t i = 0; i < K - 1; i++) {
116  endIndices[i] = startIndices[i + 1];
117  }
118 
119  // 5. Put the primitives in separate lists and return them like the API
120  // says.
121  std::array<PrimList, K> subVolumePrimitives;
122  for (size_t i = 0; i < K; i++) {
123 
124  // God how I hate how the standard library handles iterator offsets. Fuck
125  // you, long/unsigned long conversion.
126  typename PrimList::const_iterator first =
127  sortedPrimitives.cbegin() + static_cast<typename PrimList::difference_type>(startIndices[i]);
128  typename PrimList::const_iterator last =
129  sortedPrimitives.cbegin() + static_cast<typename PrimList::difference_type>(endIndices[i]);
130 
131  subVolumePrimitives[i] = PrimList(first, last);
132  }
133 
134  return subVolumePrimitives;
135  };
136 
137  // Stop function. Exists subdivision if there are not enough primitives left
138  // to keep subdividing. We set the limit at 10 primitives.
139  EBGeometry::BVH::StopFunctionT<T, P, BV, K> stopFunc = [](const BuilderNode& a_node) -> bool {
140  const size_t numPrimsInNode = (a_node.getPrimitives()).size();
141  return numPrimsInNode < K;
142  };
143 
144  // Init the root node and partition the primitives.
145  auto root = std::make_shared<BuilderNode>(a_distanceFunctions);
146 
147  root->topDownSortAndPartitionPrimitives(a_bvConstructor, partitioner, stopFunc);
148 
149  m_rootNode = root->flattenTree();
150 }

152 template <class T, class P, class BV, size_t K>
153 T
154 UnionBVH<T, P, BV, K>::value(const Vec3T<T>& a_point) const noexcept
155 {
156  const T sign = (m_flipSign) ? -1.0 : 1.0;
157 
158  // For the CSG union we select the smallest value. This means that if a
159  // point is inside one object but outside another one, we choose the
160  // inside value. By design, LinearNode is a signed distance object, but the
161  // BVH accelerator is still a valid accelerator that permits us to do all
162  // kinds of accelerated traversals. For overlapping objects there is not signed
163  // distance but there is still a CSG union, and the BVH is still a useful thing.
164  // So, we can't use LinearNode::signedDistanceFunction because it returns the
165  // closest object and not the object with the smallest value function.
166  //
167  // Fortunately, when I wrote the LinearNode accelerator I wrote it such that we can determine
168  // how to update the "shortest" distance using externally supplied criteria.
169  // So, we just update this as f = min(f1,f2,f3) etc, and prune nodes accordingly.
170  // The criteria for that are below...
171 
172  BVH::Comparator<T> unionComparator = [](const T& dmin, const std::vector<T>& primDistances) -> T {
173  T d = dmin;
174 
175  for (const auto& primDist : primDistances) {
176  d = std::min(d, primDist);
177  }
178 
179  return d;
180  };
181 
182  BVH::Pruner<T> unionPruner = [](const T& bvDist, const T& minDist) -> bool {
183  return bvDist <= 0.0 || bvDist <= minDist;
184  };
185 
186  const T value = m_rootNode->stackPrune(a_point, unionComparator, unionPruner);
187 
188  return sign * value;
189 }
