149 template <class T>
150 std::function<T(const T& a, const T& b, const T& s)> expMin = [](const T& a, const T& b, const T& s) -> T {
151  T ret = exp(-a / s) + exp(-b / s);
152 
153  return -log(ret) * s;
154 };

162 template <class T>
163 std::function<T(const T& a, const T& b, const T& s)> smoothMin = [](const T& a, const T& b, const T& s) -> T {
164  const T h = std::max(s - std::abs(a - b), 0.0) / s;
165 
166  return std::min(a, b) - 0.25 * h * h * s;
167 };
168 
175 template <class T>
176 std::function<T(const T& a, const T& b, const T& s)> smoothMax = [](const T& a, const T& b, const T& s) -> T {
177  const T h = std::max(s - std::abs(a - b), 0.0) / s;
178 
179  return std::max(a, b) + 0.25 * h * h * s;
180 };
