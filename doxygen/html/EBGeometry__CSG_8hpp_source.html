161 template <class T>
162 std::function<T(const T& a, const T& b, const T& s)> expMin = [](const T& a, const T& b, const T& s) -> T {
163  T ret = exp(-a / s) + exp(-b / s);
164 
165  return -log(ret) * s;
166 };

174 template <class T>
175 std::function<T(const T& a, const T& b, const T& s)> smoothMin = [](const T& a, const T& b, const T& s) -> T {
176  const T h = std::max(s - std::abs(a - b), 0.0) / s;
177 
178  return std::min(a, b) - 0.25 * h * h * s;
179 };
180 
187 template <class T>
188 std::function<T(const T& a, const T& b, const T& s)> smoothMax = [](const T& a, const T& b, const T& s) -> T {
189  const T h = std::max(s - std::abs(a - b), 0.0) / s;
190 
191  return std::max(a, b) + 0.25 * h * h * s;
192 };
