37  template <class T, size_t K>
38  auto equalCounts = [](const PrimitiveList<T>& a_primitives) -> std::array<PrimitiveList<T>, K> {
39  int length = a_primitives.size() / K;
40  int remain = a_primitives.size() % K;
41 
42  int begin = 0;
43  int end = 0;
44 
45  std::array<PrimitiveList<T>, K> chunks;
46 
47  for (size_t k = 0; k < K; k++) {
48  end += (remain > 0) ? length + 1 : length;
49  remain--;
50 
51  chunks[k] = PrimitiveList<T>(a_primitives.begin() + begin, a_primitives.begin() + end);
52 
53  begin = end;
54  }
55 
56  return chunks;
57  };

112  template <class T, class BV, size_t K>
113  EBGeometry::BVH::PartitionerT<EBGeometry::DCEL::FaceT<T>, BV, K> chunkPartitioner =
114  [](const PrimitiveList<T>& a_primitives) -> std::array<PrimitiveList<T>, K> {
115  Vec3T<T> lo = Vec3T<T>::max();
116  Vec3T<T> hi = -Vec3T<T>::max();
117 
118  for (const auto& p : a_primitives) {
119  lo = min(lo, p->getCentroid());
120  hi = max(hi, p->getCentroid());
121  }
122 
123  const size_t splitDir = (hi - lo).maxDir(true);
124 
125  // Sort the primitives along the above coordinate direction.
126  PrimitiveList<T> sortedPrimitives(a_primitives);
127 
128  std::sort(sortedPrimitives.begin(),
129  sortedPrimitives.end(),
130  [splitDir](const std::shared_ptr<const FaceT<T>>& f1, const std::shared_ptr<const FaceT<T>>& f2) -> bool {
131  return f1->getCentroid(splitDir) < f2->getCentroid(splitDir);
132  });
133 
134  return EBGeometry::DCEL::equalCounts<T, K>(sortedPrimitives);

145  template <class T, class BV, size_t K>
146  EBGeometry::BVH::PartitionerT<EBGeometry::DCEL::FaceT<T>, BV, K> bvPartitioner =
147  [](const PrimitiveList<T>& a_primitives) -> std::array<PrimitiveList<T>, K> {
148  Vec3T<T> lo = Vec3T<T>::max();
149  Vec3T<T> hi = -Vec3T<T>::max();
150 
151  // Pack primitives and their bounding volumes together.
152  using P = std::pair<std::shared_ptr<const EBGeometry::DCEL::FaceT<T>>, BV>;
153 
154  std::vector<P> primsAndBVs;
155 
156  for (const auto& p : a_primitives) {
157  const BV bv(p->getAllVertexCoordinates());
158 
159  primsAndBVs.emplace_back(p, bv);
160 
161  lo = min(lo, bv.getCentroid());
162  hi = max(hi, bv.getCentroid());
163  }
164 
165  const size_t splitDir = (hi - lo).maxDir(true);
166 
167  // Sort the primitives based on the centroid location of their BVs.
168  std::sort(primsAndBVs.begin(), primsAndBVs.end(), [splitDir](const P& p1, const P& p2) {
169  return (p1.second).getCentroid()[splitDir] < (p2.second).getCentroid()[splitDir];
170  });
171 
172  // Unpack the vector and partition into equal counts.
173  PrimitiveList<T> sortedPrimitives;
174  for (const auto& p : primsAndBVs) {
175  sortedPrimitives.emplace_back(p.first);
176  }
177 
178  primsAndBVs.resize(0);
179 
180  return EBGeometry::DCEL::equalCounts<T, K>(sortedPrimitives);
181  };

189  template <class T, class BV, size_t K>
190  EBGeometry::BVH::PartitionerT<EBGeometry::DCEL::FaceT<T>, BV, K> centroidPartitioner =
191  [](const PrimitiveList<T>& a_primitives) -> std::array<PrimitiveList<T>, K> {
192  Vec3T<T> lo = Vec3T<T>::max();
193  Vec3T<T> hi = -Vec3T<T>::max();
194 
195  for (const auto& p : a_primitives) {
196  lo = min(lo, p->getCentroid());
197  hi = max(hi, p->getCentroid());
198  }
199 
200  const size_t splitDir = (hi - lo).maxDir(true);
201  const T delta = (hi - lo)[splitDir] / K;
202 
203  std::array<T, K> boundsLo;
204  std::array<T, K> boundsHi;
205 
206  for (size_t k = 0; k < K; k++) {
207  boundsLo[k] = lo[splitDir] + delta * k;
208  boundsHi[k] = lo[splitDir] + delta * (k + 1);
209  }
210 
211  // Given coord, find the bin.
212  auto getBin = [&](const T& coord) -> size_t {
213  for (size_t k = 0; k < K; k++) {
214  if (coord >= boundsLo[k] && coord <= boundsHi[k]) {
215  return k;
216  }
217  }
218 
219  return K - 1;
220  };
221 
222  // Put primitives in their respective bins.
223  std::array<PrimitiveList<T>, K> chunks;
224 
225  for (const auto& p : a_primitives) {
226  const size_t k = getBin(p->getCentroid()[splitDir]);
227  chunks[k].push_back(p);
228  }
229 
230  // The centroid-based partitioner can end up with no primitives in one of the
231  // leaves (a rare case). Use a different partitioner in that case.
232  if (!(EBGeometry::DCEL::validChunks<T, K>(chunks))) {
233  chunks = EBGeometry::DCEL::chunkPartitioner<T, BV, K>(a_primitives);
234  }
235 
236  return chunks;
237  };
