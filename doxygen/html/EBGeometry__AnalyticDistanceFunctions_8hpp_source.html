400  virtual T signedDistance(const Vec3T<T>& a_point) const noexcept override {
401  // For each coordinate direction, we have delta[dir] if a_point[dir] falls between xLo and xHi. In this case
402  // delta[dir] will be the signed distance to the closest box face in the dir-direction. Otherwise, if a_point[dir] is
403  // outside the corner we have delta[dir] > 0.
404  const Vec3T<T> delta(std::max(m_loCorner[0] - a_point[0], a_point[0] - m_hiCorner[0]), // < 0 if point falls between xLo and xHi.
405  std::max(m_loCorner[1] - a_point[1], a_point[1] - m_hiCorner[1]), // < 0 if point falls between yLo and yHi.
406  std::max(m_loCorner[2] - a_point[2], a_point[2] - m_hiCorner[2])); // < 0 if point falls between zLo and zHi.
407 
408  // Note: max is max(Vec3T<T>, Vec3T<T>) and not std::max. It returns a vector with coordinate-wise largest components. Note that
409  // the first part std::min(...) is the signed distance on the inside of the box (delta will have negative components). The other
410  // part max(Vec3T<T>::zero(), ...) is for outside the box.
411  const T d = std::min(T(0.0), delta[delta.maxDir(false)]) + max(Vec3T<T>::zero(), delta).length();
412 
413  const T sign = m_flipInside ? -1.0 : 1.0;
414 
415  return sign * d;
416  }
