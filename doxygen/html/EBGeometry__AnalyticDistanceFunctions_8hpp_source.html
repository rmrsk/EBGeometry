253  virtual T
254  signedDistance(const Vec3T<T>& a_point) const noexcept override
255  {
256  // For each coordinate direction, we have delta[dir] if a_point[dir] falls
257  // between xLo and xHi. In this case delta[dir] will be the signed distance
258  // to the closest box face in the dir-direction. Otherwise, if a_point[dir]
259  // is outside the corner we have delta[dir] > 0.
260  const Vec3T<T> delta(std::max(m_loCorner[0] - a_point[0], a_point[0] - m_hiCorner[0]),
261  std::max(m_loCorner[1] - a_point[1], a_point[1] - m_hiCorner[1]),
262  std::max(m_loCorner[2] - a_point[2], a_point[2] - m_hiCorner[2]));
263 
264  // Note: max is max(Vec3T<T>, Vec3T<T>) and not std::max. It returns a
265  // vector with coordinate-wise largest components. Note that the first part
266  // std::min(...) is the signed distance on the inside of the box (delta will
267  // have negative components). The other part max(Vec3T<T>::zero(), ...) is
268  // for outside the box.
269  const T d = std::min(T(0.0), delta[delta.maxDir(false)]) + max(Vec3T<T>::zero(), delta).length();
270 
271  return d;
272  }
273 
274 protected:
278  Vec3T<T> m_loCorner;
279 
283  Vec3T<T> m_hiCorner;

478  virtual T
479  signedDistance(const Vec3T<T>& a_point) const noexcept override
480  {
481  T d = std::numeric_limits<T>::infinity();
482 
483  if (m_length > 0.0 && m_radius > 0.0) {
484  const Vec3T<T> point = a_point - m_center;
485  const T para = dot(point, m_axis);
486  const Vec3T<T> ortho = point - para * m_axis; // Distance from cylinder axis.
487 
488  const T w = ortho.length() - m_radius; // Distance from cylinder wall. < 0
489  // on inside and > 0 on outside.
490  const T h = std::abs(para) - 0.5 * m_length; // Distance from cylinder top. < 0 on
491  // inside and > 0 on outside.
492 
493  constexpr T zero = T(0.0);
494 
495  if (w <= zero && h <= zero) { // Inside cylinder
496  d = (std::abs(w) < std::abs(h)) ? w : h;
497  }
498  else if (w <= zero && h > zero) { // Above one of the endcaps.
499  d = h;
500  }
501  else if (w > zero && h < zero) { // Outside radius but between the
502  // endcaps.
503  d = w;
504  }
505  else {
506  d = sqrt(w * w + h * h);
507  }
508  }
509 
510  return d;
511  }

517  Vec3T<T> m_center1;
518

762  {
763  const Vec3T<T> delta = a_point - m_tip;
764  const T dr = sqrt(delta[0] * delta[0] + delta[1] * delta[1]);
765  const T dz = delta[2];
766

768  constexpr T one = T(1.0);
769 
770  const Vec2T<T> q = m_height * Vec2T<T>(m_c.x / m_c.y, -1.0);
771  const Vec2T<T> w = Vec2T<T>(dr, dz);
772  const Vec2T<T> a = w - clamp(dot(w, q) / dot(q, q), zero, one) * q;
773  const Vec2T<T> b = w - Vec2T<T>(q.x * clamp(w.x / q.x, zero, one), q.y);
774 
775  auto sign = [](const T& x) { return (x > zero) - (x < zero); };
776 
777  const T k = sign(q.y);
778  const T d = std::min(dot(a, a), dot(b, b));
779  const T s = std::max(k * (w.x * q.y - w.y * q.x), k * (w.y - q.y));
780 
781  return sqrt(d) * sign(s);
782  }
