427  virtual T
428  signedDistance(const Vec3T<T>& a_point) const noexcept override
429  {
430  // For each coordinate direction, we have delta[dir] if a_point[dir] falls
431  // between xLo and xHi. In this case delta[dir] will be the signed distance
432  // to the closest box face in the dir-direction. Otherwise, if a_point[dir]
433  // is outside the corner we have delta[dir] > 0.
434  const Vec3T<T> delta(std::max(m_loCorner[0] - a_point[0],
435  a_point[0] - m_hiCorner[0]), // < 0 if point falls between xLo and xHi.
436  std::max(m_loCorner[1] - a_point[1],
437  a_point[1] - m_hiCorner[1]), // < 0 if point falls between yLo and yHi.
438  std::max(m_loCorner[2] - a_point[2],
439  a_point[2] - m_hiCorner[2])); // < 0 if point falls between zLo and zHi.
440 
441  // Note: max is max(Vec3T<T>, Vec3T<T>) and not std::max. It returns a
442  // vector with coordinate-wise largest components. Note that the first part
443  // std::min(...) is the signed distance on the inside of the box (delta will
444  // have negative components). The other part max(Vec3T<T>::zero(), ...) is
445  // for outside the box.
446  const T d = std::min(T(0.0), delta[delta.maxDir(false)]) + max(Vec3T<T>::zero(), delta).length();
447 
448  const T sign = m_flipInside ? -1.0 : 1.0;
449 
450  return sign * d;
451  }
