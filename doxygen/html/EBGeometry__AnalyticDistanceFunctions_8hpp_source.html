423  virtual T
424  signedDistance(const Vec3T<T>& a_point) const noexcept override
425  {
426  // For each coordinate direction, we have delta[dir] if a_point[dir] falls
427  // between xLo and xHi. In this case delta[dir] will be the signed distance
428  // to the closest box face in the dir-direction. Otherwise, if a_point[dir]
429  // is outside the corner we have delta[dir] > 0.
430  const Vec3T<T> delta(std::max(m_loCorner[0] - a_point[0],
431  a_point[0] - m_hiCorner[0]), // < 0 if point falls between xLo and xHi.
432  std::max(m_loCorner[1] - a_point[1],
433  a_point[1] - m_hiCorner[1]), // < 0 if point falls between yLo and yHi.
434  std::max(m_loCorner[2] - a_point[2],
435  a_point[2] - m_hiCorner[2])); // < 0 if point falls between zLo and zHi.
436 
437  // Note: max is max(Vec3T<T>, Vec3T<T>) and not std::max. It returns a
438  // vector with coordinate-wise largest components. Note that the first part
439  // std::min(...) is the signed distance on the inside of the box (delta will
440  // have negative components). The other part max(Vec3T<T>::zero(), ...) is
441  // for outside the box.
442  const T d = std::min(T(0.0), delta[delta.maxDir(false)]) + max(Vec3T<T>::zero(), delta).length();
443 
444  const T sign = m_flipInside ? -1.0 : 1.0;
445 
446  return sign * d;
447  }
